----------------tablas-------------------

CREATE TABLE usuarios
(id: VARCHAR(20) NOT NULL,
nombre: VARCHAR(50) NOT NULL,
correo VARCHAR(80) NOT NULL,
genero: VARCHAR(10) NOT NULL,
telefono: VARCHAR(15) NOT NULL,
nacimiento: DATE NOT NULL,
password: VARCHAR(30) NOT NULL
);

CREATE TABLE clientes
(id_usuario: VARCHAR(20) NOT NULL,
user_name: VARCHAR(50) NOT NULL
);

CREATE TABLE asociaciones
(id_usuario: VARCHAR(20) NOT NULL,
nit: VARCHAR(15) NOT NULL,
razon_social: VARCHAR(30) NOT NULL,
sede: VARCHAR(30) NOT NULL
);

CREATE TABLE objetos
(id: NUMBER(10) NOT NULL,
nombre: VARCHAR(50) NOT NULL,
descripcion: VARCHAR(100) NOT NULL,
tipo: VARCHAR(20) NOT NULL,
id_usuario: VARCHAR(20) NOT NULL
);

CREATE TABLE imagenes
(id_objeto: NUMBER(10) NOT NULL,
imagen: VARCHAR(50) NOT NULL,
id_imagen: NUMBER(10) NOT NULL
);

CREATE TABLE publicaciones
(id: NUMBER(10) NOT NULL,
valoracion: VARCHAR(20) NOT NULL,
fecha_creacion: DATE NOT NULL,
estado: BOOLEAN NOT NULL,
fecha_terminacion: DATE NOT NULL,
id_usuario: VARCHAR(20) NOT NULL
);

CREATE TABLE participaciones
(id_usuario: VARCHAR(20) NOT NULL,
id_publicacion: NUMBER(10) NOT NULL,
);

CREATE TABLE reciclajes
(id_publicacion: VARCHAR(20) NOT NULL,
fecha: DATE NOT NULL,
nit_asociacion: VARCHAR(15) NOT NULL,
);

CREATE TABLE fiestas
(id_publicacion: VARCHAR(20) NOT NULL,
fecha_inicio: DATE NOT NULL,
fecha_fin: DATE NOT NULL,
n_participantes: NUMBER(5) NOT NULL,
id_actividad: NUMBER(5) NOT NULL
);
